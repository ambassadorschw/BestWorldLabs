package main

import "fmt" //Импортируем пакет fmt для работы с форматированием ввода-вывода  

func main() { 

	fmt.Println("Задание 1:")
	var num int //Объявляем переменную num типа int для хранения введенного числа  
	fmt.Scan(&num) //Сканируем (читаем) введенное пользователем число и сохраняем его в переменной num  
	if num%2 == 0 { //Проверяем, является ли число четным (остаток от деления на 2 равен 0)  
		fmt.Println("Число четное")
	} else {
		fmt.Println("Число нечетное")
	}

}
